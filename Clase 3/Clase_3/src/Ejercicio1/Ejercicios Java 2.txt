import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;

public class OperacionesConArchivo {
    public static void main(String[] args) {
        if (args.length != 2) {
            System.out.println("Uso incorrecto. Debe proporcionar la ruta del archivo y la operación (suma/multiplicacion).");
            return;
        }

        String rutaArchivo = args[0];
        String operacion = args[1].toLowerCase();

        try {
            BufferedReader reader = new BufferedReader(new FileReader(rutaArchivo));
            String linea;
            double resultado = operacion.equals("suma") ? 0.0 : 1.0;

            while ((linea = reader.readLine()) != null) {
                double numero = Double.parseDouble(linea);
                if (operacion.equals("suma")) {
                    resultado += numero;
                } else if (operacion.equals("multiplicacion")) {
                    resultado *= numero;
                } else {
                    System.out.println("Operación no válida. Debe ser 'suma' o 'multiplicacion'.");
                    return;
                }
            }

            reader.close();

            System.out.println("Resultado de " + operacion + " de los números en el archivo: " + resultado);
        } catch (IOException e) {
            System.out.println("Error al leer el archivo: " + e.getMessage());
        } catch (NumberFormatException e) {
            System.out.println("El archivo contiene datos no numéricos: " + e.getMessage());
        }
    }
}
