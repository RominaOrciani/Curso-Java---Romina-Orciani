public class CifradoCesar {
    private static final String ABECEDARIO = "abcdefghijklmnñopqrstuvwxyz ";

    public static String codificar(String mensaje, int desplazamiento) {
        StringBuilder resultado = new StringBuilder();
        for (char caracter : mensaje.toCharArray()) {
            int indice = ABECEDARIO.indexOf(caracter);
            if (indice != -1) {
                int nuevoIndice = (indice + desplazamiento) % ABECEDARIO.length();
                resultado.append(ABECEDARIO.charAt(nuevoIndice));
            } else {
                resultado.append(caracter); // Conservar caracteres no encontrados en el abecedario
            }
        }
        return resultado.toString();
    }

    public static String decodificar(String mensaje, int desplazamiento) {
        // La decodificación es igual que la codificación, pero con desplazamiento negativo
        return codificar(mensaje, -desplazamiento);
    }

    public static void main(String[] args) {
        String mensaje = "hola que tal";
        int desplazamiento = 1;

        String mensajeCodificado = codificar(mensaje, desplazamiento);
        String mensajeDecodificado = decodificar(mensajeCodificado, desplazamiento);

        System.out.println("Mensaje Original: " + mensaje);
        System.out.println("Mensaje Codificado: " + mensajeCodificado);
        System.out.println("Mensaje Decodificado: " + mensajeDecodificado);
    }
}

public class SumaNumerosMayoresQueX {
    public static int sumaNumerosMayoresQueX(int[] numeros, int x) {
        int suma = 0;
        for (int numero : numeros) {
            if (numero > x) {
                suma += numero;
            }
        }
        return suma;
    }

    public static void main(String[] args) {
        int[] numeros = {1, 5, 10, 15, 20, 25};
        int x = 10;

        int resultado = sumaNumerosMayoresQueX(numeros, x);
        System.out.println("La suma de los números mayores que " + x + " es: " + resultado);
    }
}
